[docs]def password_hash(password: Password) -> Combo:
    """
    Generate hash for password
    """
    salt = __os.urandom(32)
    key = __hash.pbkdf2_hmac(
        hash_name="sha256",
        password=password.encode("utf-8"),
        salt=salt,
        iterations=100000,
    )
    out = {
        "salt": salt,
        "key": key,
    }
    return out



[docs]def password_hash_check(
        password: Password,
        combo: Combo,
    ) -> bool:
    """
    Compare hashes between 2 passwords
    """
    if "salt" in combo and "key" in combo:
        salt = combo["salt"]
        compare_key = combo["key"]
    else:
        return None
    
    key = __hash.pbkdf2_hmac(
        hash_name="sha256",
        password=password.encode("utf-8"),
        salt=salt,
        iterations=100000,
    )

    if key==compare_key:
        return True
    else:
        return False
